n = int(input("Nhập n: "))
S = 0

for i in range(1, n + 1):
    S += 1 / i

print("S =", S)



#2b
n = int(input("Nhập n: "))
S = 0

for i in range(1, n + 1):
    S += 1 / (i * (i + 1))

print("S =", S)

import math

n = int(input("Nhập n: "))
S = 0

for i in range(2, n + 2):
    S += 1 / math.sqrt(i)

print("S =", S)
