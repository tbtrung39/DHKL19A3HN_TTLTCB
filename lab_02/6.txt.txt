n = int(input("Nhập số có 3 chữ số: "))

if n < 100 or n > 999:
    print("Số không hợp lệ")
else:
    so = ["không", "một", "hai", "ba", "bốn", "năm", "sáu", "bảy", "tám", "chín"]

    tram = n // 100
    chuc = (n // 10) % 10
    donvi = n % 10

    print(so[tram], "trăm", end=" ")

    if chuc == 0 and donvi != 0:
        print("lẻ", so[donvi])
    elif chuc == 1:
        print("mười", so[donvi])
    else:
        print(so[chuc], "mươi", so[donvi])
