s = input("Nhập chuỗi: ").upper()

hex_chars = "0123456789ABCDEF"
hop_le = True

for c in s:
    if c not in hex_chars:
        hop_le = False

if hop_le:
    decimal = int(s, 16)
    print("Số thập phân:", decimal)
else:
    new_s = ""
    for c in s:
        if c in hex_chars:
            new_s += c
    
    if new_s == "":
        print("Không có ký tự Hex hợp lệ")
    else:
        print("Chuỗi sau khi lọc:", new_s)
        print("Số thập phân:", int(new_s, 16))
