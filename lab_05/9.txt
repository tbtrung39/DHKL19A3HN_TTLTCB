s = input("Nhập chuỗi: ")

max_sub = ""
current = s[0]

for i in range(1, len(s)):
    if s[i] == s[i-1]:
        current += s[i]
    else:
        if len(current) > len(max_sub):
            max_sub = current
        current = s[i]


if len(current) > len(max_sub):
    max_sub = current

print("Chuỗi con dài nhất:", max_sub)
